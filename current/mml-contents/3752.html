definition
let Al be QC-alphabet ;
let X be Subset of (CQC-WFF Al);
attr X is negation_faithful means :Def1: :: GOEDELCP:def 1
 for p being Element of CQC-WFF Al holds
( X |- p or X |- 'not' p );
end;
